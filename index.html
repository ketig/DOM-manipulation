­Ъџђ JavaScript DOM рЃЏрЃљрЃюрЃўрЃърЃБрЃџрЃљрЃфрЃўрЃљ

­ЪЊІ рЃАрЃћрЃАрЃўрЃўрЃА рЃњрЃћрЃњрЃЏрЃљ

­Ъј» рЃЏрЃўрЃќрЃљрЃюрЃў

JavaScript-рЃўрЃА рЃњрЃљрЃЏрЃЮрЃДрЃћрЃюрЃћрЃЉрЃўрЃЌ DOM-рЃўрЃА рЃЏрЃљрЃюрЃўрЃърЃБрЃџрЃљрЃфрЃўрЃљ рЃЊрЃљ рЃўрЃюрЃбрЃћрЃарЃљрЃЦрЃбрЃўрЃБрЃџрЃў рЃЋрЃћрЃЉ рЃњрЃЋрЃћрЃарЃЊрЃћрЃЉрЃўрЃА рЃерЃћрЃЦрЃЏрЃюрЃљ.

­ЪЊџ рЃАрЃгрЃљрЃЋрЃџрЃўрЃА рЃерЃћрЃЊрЃћрЃњрЃћрЃЉрЃў

  • DOM-рЃўрЃА рЃАрЃбрЃарЃБрЃЦрЃбрЃБрЃарЃўрЃА рЃњрЃљрЃњрЃћрЃЉрЃљ
  • рЃћрЃџрЃћрЃЏрЃћрЃюрЃбрЃћрЃЉрЃўрЃА рЃЏрЃЮрЃФрЃўрЃћрЃЉрЃљ рЃЊрЃљ рЃерЃћрЃЦрЃЏрЃюрЃљ
  • Attributes-рЃћрЃЉрЃўрЃА рЃЏрЃљрЃюрЃўрЃърЃБрЃџрЃљрЃфрЃўрЃљ
  • Events-рЃћрЃЉрЃўрЃА рЃЏрЃљрЃарЃЌрЃЋрЃљ
  • рЃўрЃюрЃбрЃћрЃарЃљрЃЦрЃбрЃўрЃБрЃџрЃў рЃўрЃюрЃбрЃћрЃарЃцрЃћрЃўрЃАрЃћрЃЉрЃўрЃА рЃерЃћрЃЦрЃЏрЃюрЃљ

РЈ░ рЃЊрЃарЃЮрЃўрЃА рЃњрЃљрЃюрЃљрЃгрЃўрЃџрЃћрЃЉрЃљ (90 рЃгрЃБрЃЌрЃў)

  • DOM рЃњрЃљрЃњрЃћрЃЉрЃљ: 15 рЃгрЃБрЃЌрЃў
  • Script рЃўрЃюрЃбрЃћрЃњрЃарЃљрЃфрЃўрЃљ: 10 рЃгрЃБрЃЌрЃў
  • рЃћрЃџрЃћрЃЏрЃћрЃюрЃбрЃћрЃЉрЃўрЃА рЃЏрЃљрЃюрЃўрЃърЃБрЃџрЃљрЃфрЃўрЃљ: 20 рЃгрЃБрЃЌрЃў
  • Query Selectors: 10 рЃгрЃБрЃЌрЃў
  • Events: 20 рЃгрЃБрЃЌрЃў
  • рЃърЃарЃљрЃЦрЃбрЃўрЃЎрЃБрЃџрЃў рЃЊрЃљрЃЋрЃљрЃџрЃћрЃЉрЃћрЃЉрЃў: 15 рЃгрЃБрЃЌрЃў

­Ъїљ DOM рЃњрЃљрЃњрЃћрЃЉрЃљ

­Ъцћ рЃарЃљ рЃљрЃарЃўрЃА DOM?

DOM (Document Object Model) - HTML рЃЊрЃЮрЃЎрЃБрЃЏрЃћрЃюрЃбрЃўрЃА рЃ«рЃўрЃА рЃАрЃбрЃарЃБрЃЦрЃбрЃБрЃарЃљ, рЃарЃЮрЃЏрЃћрЃџрЃўрЃф JavaScript-рЃА рЃАрЃљрЃерЃБрЃљрЃџрЃћрЃЉрЃљрЃА рЃљрЃФрЃџрЃћрЃЋрЃА рЃЋрЃћрЃЉ рЃњрЃЋрЃћрЃарЃЊрЃЌрЃљрЃю рЃўрЃюрЃбрЃћрЃарЃљрЃЦрЃфрЃўрЃўрЃАрЃЌрЃЋрЃўрЃА.

рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃў: DOM рЃ«рЃћ

HTML:
<html>
  <head>
    <title>рЃЕрЃћрЃЏрЃў рЃњрЃЋрЃћрЃарЃЊрЃў</title>
  </head>
  <body>
    <h1>рЃАрЃљрЃЌрЃљрЃБрЃарЃў</h1>
    <p>рЃърЃљрЃарЃљрЃњрЃарЃљрЃцрЃў</p>
  </body>
</html>

DOM рЃ«рЃћ:
document
  РћћРћђ html
      РћюРћђ head
      Рћѓ   РћћРћђ title
      РћћРћђ body
          РћюРћђ h1
          РћћРћђ p

­ЪћЉ рЃФрЃўрЃарЃўрЃЌрЃљрЃЊрЃў рЃфрЃюрЃћрЃЉрЃћрЃЉрЃў

  • Node: DOM-рЃўрЃА рЃДрЃЋрЃћрЃџрЃљ рЃћрЃџрЃћрЃЏрЃћрЃюрЃбрЃў
  • Element: HTML рЃбрЃћрЃњрЃћрЃЉрЃў
  • Text Node: рЃбрЃћрЃЦрЃАрЃбрЃБрЃарЃў рЃерЃўрЃюрЃљрЃљрЃарЃАрЃў
  • Attribute: рЃћрЃџрЃћрЃЏрЃћрЃюрЃбрЃўрЃА рЃЌрЃЋрЃўрЃАрЃћрЃЉрЃћрЃЉрЃў

­ЪЊю Script рЃўрЃюрЃбрЃћрЃњрЃарЃљрЃфрЃўрЃљ

­ЪћД JavaScript-рЃўрЃА HTML-рЃерЃў рЃЕрЃљрЃарЃЌрЃЋрЃўрЃА рЃњрЃќрЃћрЃЉрЃў

1. Inline Script:

<script>
    console.log("рЃњрЃљрЃЏрЃљрЃарЃ»рЃЮрЃЉрЃљ!");
</script>

2. External Script:

<script src="script.js"></script>

3. Event Attribute:

<button onclick="alert('рЃдрЃўрЃџрЃљрЃЎрЃќрЃћ рЃЊрЃљрЃгрЃЎрЃљрЃърЃБрЃюрЃћрЃЉрЃљ!')">рЃЊрЃљрЃљрЃГрЃўрЃарЃћ</button>

­ЪЊЇ Script-рЃўрЃА рЃњрЃљрЃюрЃЌрЃљрЃЋрЃАрЃћрЃЉрЃўрЃА рЃљрЃЊрЃњрЃўрЃџрЃћрЃЉрЃў

  • <head> рЃАрЃћрЃЦрЃфрЃўрЃљрЃерЃў: рЃЏрЃЮрЃЏрЃќрЃљрЃЊрЃћрЃЉрЃљ рЃњрЃЋрЃћрЃарЃЊрЃўрЃА рЃЕрЃљрЃбрЃЋрЃўрЃарЃЌрЃЋрЃљрЃЏрЃЊрЃћ
  • <body>-рЃўрЃА рЃЉрЃЮрЃџрЃЮрЃерЃў: HTML-рЃўрЃА рЃЕрЃљрЃбрЃЋрЃўрЃарЃЌрЃЋрЃўрЃА рЃерЃћрЃЏрЃЊрЃћрЃњ
  • DOMContentLoaded: DOM-рЃўрЃА рЃЏрЃќрЃљрЃЊрЃДрЃЮрЃцрЃюрЃўрЃА рЃерЃћрЃЏрЃЮрЃгрЃЏрЃћрЃЉрЃљ

DOMContentLoaded рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃў:

document.addEventListener('DOMContentLoaded', function() {
    console.log('DOM рЃЏрЃќрЃљрЃЊрЃљрЃљ!');
    // рЃЎрЃЮрЃЊрЃў рЃњрЃљрЃћрЃерЃЋрЃћрЃЉрЃљ DOM-рЃўрЃА рЃЕрЃљрЃбрЃЋрЃўрЃарЃЌрЃЋрЃўрЃА рЃерЃћрЃЏрЃЊрЃћрЃњ
});

РџА Defer vs Async

­Ъћё Script-рЃўрЃА рЃЕрЃљрЃбрЃЋрЃўрЃарЃЌрЃЋрЃўрЃА рЃарЃћрЃЪрЃўрЃЏрЃћрЃЉрЃў

рЃАрЃбрЃљрЃюрЃЊрЃљрЃарЃбрЃБрЃџрЃў рЃЕрЃљрЃбрЃЋрЃўрЃарЃЌрЃЋрЃљ:

<script src="script.js"></script>
// HTML parsing рЃАрЃЕрЃћрЃарЃЊрЃћрЃЉрЃљ, Script рЃЕрЃљрЃўрЃбрЃЋрЃўрЃарЃЌрЃћрЃЉрЃљ рЃЊрЃљ рЃњрЃљрЃћрЃерЃЋрЃћрЃЉрЃљ

Async Attribute:

<script src="script.js" async></script>
// HTML parsing рЃЊрЃљ Script рЃЕрЃљрЃбрЃЋрЃўрЃарЃЌрЃЋрЃљ рЃърЃљрЃарЃљрЃџрЃћрЃџрЃБрЃарЃљрЃЊ рЃ«рЃЊрЃћрЃЉрЃљ
// Script рЃњрЃљрЃћрЃерЃЋрЃћрЃЉрЃљ рЃЕрЃљрЃбрЃЋрЃўрЃарЃЌрЃЋрЃўрЃА рЃерЃћрЃЏрЃЊрЃћрЃњ (HTML parsing рЃерЃћрЃўрЃФрЃџрЃћрЃЉрЃљ рЃерЃћрЃЕрЃћрЃарЃЊрЃћрЃА)

Defer Attribute:

<script src="script.js" defer></script>
// HTML parsing рЃЊрЃљ Script рЃЕрЃљрЃбрЃЋрЃўрЃарЃЌрЃЋрЃљ рЃърЃљрЃарЃљрЃџрЃћрЃџрЃБрЃарЃљрЃЊ рЃ«рЃЊрЃћрЃЉрЃљ
// Script рЃњрЃљрЃћрЃерЃЋрЃћрЃЉрЃљ HTML parsing-рЃўрЃА рЃЊрЃљрЃАрЃарЃБрЃџрЃћрЃЉрЃўрЃА рЃерЃћрЃЏрЃЊрЃћрЃњ

­ЪЊі рЃарЃЮрЃЊрЃўрЃА рЃарЃљрЃА рЃњрЃљрЃЏрЃЮрЃЋрЃўрЃДрЃћрЃюрЃЮрЃЌ?

  • Defer: рЃарЃЮрЃфрЃљ Script-рЃА DOM рЃГрЃўрЃарЃЊрЃћрЃЉрЃљ (рЃДрЃЋрЃћрЃџрЃљрЃќрЃћ рЃ«рЃерЃўрЃарЃў)
  • Async: рЃЊрЃљрЃЏрЃЮрЃБрЃЎрЃўрЃЊрЃћрЃЉрЃћрЃџрЃў Script-рЃћрЃЉрЃў (Analytics, Ads)
  • рЃарЃЮрЃфрЃљ рЃљрЃа рЃљрЃарЃўрЃА Attribute: рЃЎрЃарЃўрЃбрЃўрЃЎрЃБрЃџрЃў Script-рЃћрЃЉрЃў

­Ъће рЃћрЃџрЃћрЃЏрЃћрЃюрЃбрЃћрЃЉрЃўрЃА рЃерЃћрЃЦрЃЏрЃюрЃљ

­ЪЈЌ№ИЈ рЃљрЃ«рЃљрЃџрЃў рЃћрЃџрЃћрЃЏрЃћрЃюрЃбрЃўрЃА рЃерЃћрЃЦрЃЏрЃюрЃљ

createElement рЃЏрЃћрЃЌрЃЮрЃЊрЃў:

// рЃљрЃ«рЃљрЃџрЃў рЃърЃљрЃарЃљрЃњрЃарЃљрЃцрЃўрЃА рЃерЃћрЃЦрЃЏрЃюрЃљ
const newParagraph = document.createElement('p');
newParagraph.textContent = 'рЃћрЃА рЃљрЃарЃўрЃА рЃљрЃ«рЃљрЃџрЃў рЃърЃљрЃарЃљрЃњрЃарЃљрЃцрЃў';

// рЃњрЃЋрЃћрЃарЃЊрЃќрЃћ рЃЊрЃљрЃЏрЃљрЃбрЃћрЃЉрЃљ
document.body.appendChild(newParagraph);

innerHTML рЃњрЃљрЃЏрЃЮрЃДрЃћрЃюрЃћрЃЉрЃљ:

const container = document.getElementById('container');
container.innerHTML = `
    <h2>рЃљрЃ«рЃљрЃџрЃў рЃАрЃљрЃЌрЃљрЃБрЃарЃў</h2>
    <p>рЃљрЃ«рЃљрЃџрЃў рЃърЃљрЃарЃљрЃњрЃарЃљрЃцрЃў</p>
    <button onclick="alert('рЃЏрЃБрЃерЃљрЃЮрЃЉрЃА!')">рЃдрЃўрЃџрЃљрЃЎрЃў</button>
`;

­ЪЊЇ рЃћрЃџрЃћрЃЏрЃћрЃюрЃбрЃўрЃА рЃњрЃљрЃюрЃЌрЃљрЃЋрЃАрЃћрЃЉрЃљ

  • appendChild(): рЃЉрЃЮрЃџрЃЮ рЃерЃЋрЃўрЃџрЃљрЃЊ рЃЊрЃљрЃЏрЃљрЃбрЃћрЃЉрЃљ
  • insertBefore(): рЃЎрЃЮрЃюрЃЎрЃарЃћрЃбрЃБрЃџрЃў рЃћрЃџрЃћрЃЏрЃћрЃюрЃбрЃўрЃА рЃгрЃўрЃю
  • replaceChild(): рЃћрЃџрЃћрЃЏрЃћрЃюрЃбрЃўрЃА рЃЕрЃљрЃюрЃљрЃфрЃЋрЃџрЃћрЃЉрЃљ
  • removeChild(): рЃћрЃџрЃћрЃЏрЃћрЃюрЃбрЃўрЃА рЃгрЃљрЃерЃџрЃљ

рЃърЃарЃљрЃЦрЃбрЃўрЃЎрЃБрЃџрЃў рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃў:

// рЃљрЃ«рЃљрЃџрЃў рЃџрЃўрЃАрЃбрЃўрЃА рЃерЃћрЃЦрЃЏрЃюрЃљ
const list = document.createElement('ul');

// рЃџрЃўрЃАрЃбрЃўрЃА рЃћрЃџрЃћрЃЏрЃћрЃюрЃбрЃћрЃЉрЃўрЃА рЃЊрЃљрЃЏрЃљрЃбрЃћрЃЉрЃљ
const items = ['рЃЋрЃљрЃерЃџрЃў', 'рЃЉрЃљрЃюрЃљрЃюрЃў', 'рЃДрЃБрЃарЃФрЃћрЃюрЃў'];
items.forEach(item => {
    const li = document.createElement('li');
    li.textContent = item;
    list.appendChild(li);
});

// рЃњрЃЋрЃћрЃарЃЊрЃќрЃћ рЃЊрЃљрЃЏрЃљрЃбрЃћрЃЉрЃљ
document.body.appendChild(list);

­ЪЈи№ИЈ Attributes рЃЏрЃљрЃюрЃўрЃърЃБрЃџрЃљрЃфрЃўрЃљ

­ЪћД Attributes-рЃћрЃЉрЃўрЃА рЃЏрЃљрЃарЃЌрЃЋрЃљ

getAttribute рЃЊрЃљ setAttribute:

const image = document.querySelector('img');

// Attribute-рЃўрЃА рЃЏрЃЮрЃФрЃўрЃћрЃЉрЃљ
const src = image.getAttribute('src');
console.log(src);

// Attribute-рЃўрЃА рЃЊрЃљрЃДрЃћрЃюрЃћрЃЉрЃљ
image.setAttribute('src', 'new-image.jpg');
image.setAttribute('alt', 'рЃљрЃ«рЃљрЃџрЃў рЃАрЃБрЃарЃљрЃЌрЃў');

// Attribute-рЃўрЃА рЃгрЃљрЃерЃџрЃљ
image.removeAttribute('title');

рЃърЃўрЃарЃЊрЃљрЃърЃўрЃарЃў Attribute рЃгрЃЋрЃЊрЃЮрЃЏрЃљ:

const input = document.querySelector('input');

// рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЉрЃўрЃА рЃЏрЃўрЃдрЃћрЃЉрЃљ рЃЊрЃљ рЃЊрЃљрЃДрЃћрЃюрЃћрЃЉрЃљ
console.log(input.value);
input.value = 'рЃљрЃ«рЃљрЃџрЃў рЃбрЃћрЃЦрЃАрЃбрЃў';

// Boolean Attributes
input.disabled = true;
input.checked = false;

// CSS Class-рЃћрЃЉрЃўрЃА рЃЏрЃљрЃюрЃўрЃърЃБрЃџрЃљрЃфрЃўрЃљ
input.className = 'form-control active';

­Ъје CSS Classes-рЃћрЃЉрЃўрЃА рЃЏрЃљрЃарЃЌрЃЋрЃљ

classList API:

const element = document.querySelector('.my-element');

// Class-рЃўрЃА рЃЊрЃљрЃЏрЃљрЃбрЃћрЃЉрЃљ
element.classList.add('active');
element.classList.add('highlighted', 'important');

// Class-рЃўрЃА рЃгрЃљрЃерЃџрЃљ
element.classList.remove('inactive');

// Class-рЃўрЃА toggle
element.classList.toggle('hidden');

// Class-рЃўрЃА рЃерЃћрЃЏрЃЮрЃгрЃЏрЃћрЃЉрЃљ
if (element.classList.contains('active')) {
    console.log('рЃћрЃџрЃћрЃЏрЃћрЃюрЃбрЃў рЃљрЃЦрЃбрЃўрЃБрЃарЃўрЃљ');
}

­ЪћЇ Query Selectors

­Ъј» рЃћрЃџрЃћрЃЏрЃћрЃюрЃбрЃћрЃЉрЃўрЃА рЃЏрЃЮрЃФрЃўрЃћрЃЉрЃљ

рЃФрЃўрЃарЃўрЃЌрЃљрЃЊрЃў Selectors:

// ID-рЃўрЃЌ рЃЏрЃЮрЃФрЃўрЃћрЃЉрЃљ
const header = document.getElementById('header');

// Class-рЃўрЃЌ рЃЏрЃЮрЃФрЃўрЃћрЃЉрЃљ
const buttons = document.getElementsByClassName('btn');

// Tag-рЃўрЃЌ рЃЏрЃЮрЃФрЃўрЃћрЃЉрЃљ
const paragraphs = document.getElementsByTagName('p');

// CSS Selector-рЃўрЃЌ (рЃърЃўрЃарЃЋрЃћрЃџрЃў рЃЏрЃљрЃбрЃЕрЃў)
const firstButton = document.querySelector('.btn');
const navList = document.querySelector('#nav ul');

// CSS Selector-рЃўрЃЌ (рЃДрЃЋрЃћрЃџрЃљ рЃЏрЃљрЃбрЃЕрЃў)
const allButtons = document.querySelectorAll('.btn');
const activeElements = document.querySelectorAll('.active');

­Ъћё рЃарЃЌрЃБрЃџрЃў Selectors

Advanced Selectors:

// Attribute Selectors

// Pseudo-class Selectors

РџА NodeList vs HTMLCollection

  • NodeList: querySelectorAll() рЃљрЃЉрЃарЃБрЃюрЃћрЃЉрЃА, forEach() рЃЏрЃћрЃЌрЃЮрЃЊрЃў рЃљрЃЦрЃЋрЃА
  • HTMLCollection: getElementsBy* рЃЏрЃћрЃЌрЃЮрЃЊрЃћрЃЉрЃў рЃљрЃЉрЃарЃБрЃюрЃћрЃЉрЃћрЃю, Live Collection

NodeList-рЃќрЃћ рЃЏрЃБрЃерЃљрЃЮрЃЉрЃљ:

const buttons = document.querySelectorAll('.btn');

// forEach рЃњрЃљрЃЏрЃЮрЃДрЃћрЃюрЃћрЃЉрЃљ
buttons.forEach(button => {
    button.addEventListener('click', function() {
        console.log('рЃдрЃўрЃџрЃљрЃЎрЃќрЃћ рЃЊрЃљрЃгрЃЎрЃљрЃърЃБрЃюрЃћрЃЉрЃљ!');
    });
});

// Array-рЃљрЃЊ рЃњрЃљрЃЊрЃљрЃДрЃЋрЃљрЃюрЃљ
const buttonsArray = Array.from(buttons);

­Ъјф Events рЃАрЃљрЃцрЃБрЃФрЃЋрЃџрЃћрЃЉрЃў

­ЪјГ рЃарЃљ рЃљрЃарЃўрЃА Event?

Event рЃљрЃарЃўрЃА рЃЏрЃЮрЃЋрЃџрЃћрЃюрЃљ, рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃ«рЃЊрЃћрЃЉрЃљ рЃЋрЃћрЃЉ рЃњрЃЋрЃћрЃарЃЊрЃќрЃћ - рЃЏрЃљрЃњ. click, scroll, keypress, load рЃЊрЃљ рЃАрЃ«рЃЋрЃљ.

­ЪЊЮ рЃДрЃЋрЃћрЃџрЃљрЃќрЃћ рЃ«рЃерЃўрЃарЃў Events

  • click: рЃдрЃўрЃџрЃљрЃЎрЃќрЃћ рЃљрЃю рЃћрЃџрЃћрЃЏрЃћрЃюрЃбрЃќрЃћ рЃЊрЃљрЃгрЃЎрЃљрЃърЃБрЃюрЃћрЃЉрЃљ
  • mouseover/mouseout: рЃЏрЃљрЃБрЃАрЃўрЃА рЃерЃћрЃАрЃЋрЃџрЃљ/рЃњрЃљрЃАрЃЋрЃџрЃљ
  • keydown/keyup: рЃЎрЃџрЃљрЃЋрЃўрЃерЃћрЃЉрЃўрЃА рЃЊрЃљрЃГрЃћрЃарЃљ/рЃњрЃљрЃерЃЋрЃћрЃЉрЃљ
  • submit: рЃцрЃЮрЃарЃЏрЃўрЃА рЃњрЃљрЃњрЃќрЃљрЃЋрЃюрЃљ
  • load: рЃњрЃЋрЃћрЃарЃЊрЃўрЃА рЃЕрЃљрЃбрЃЋрЃўрЃарЃЌрЃЋрЃљ
  • resize: рЃцрЃљрЃюрЃ»рЃарЃўрЃА рЃќрЃЮрЃЏрЃўрЃА рЃерЃћрЃфрЃЋрЃџрЃљ

addEventListener рЃЏрЃћрЃЌрЃЮрЃЊрЃў:

const button = document.querySelector('#myButton');

// Event Listener-рЃўрЃА рЃЊрЃљрЃЏрЃљрЃбрЃћрЃЉрЃљ
button.addEventListener('click', function() {
    console.log('рЃдрЃўрЃџрЃљрЃЎрЃќрЃћ рЃЊрЃљрЃгрЃЎрЃљрЃърЃБрЃюрЃЊрЃљ!');
});

// Arrow Function-рЃўрЃЌ
button.addEventListener('click', () => {
    alert('Hello World!');
});

// Named Function-рЃўрЃЌ
function handleClick() {
    console.log('Button clicked!');
}
button.addEventListener('click', handleClick);

Event Object:

button.addEventListener('click', function(event) {
    console.log('Event Type:', event.type); // click
    console.log('Target Element:', event.target); // clicked button
    console.log('Mouse Position:', event.clientX, event.clientY); // mouse coordinates
});

­Ъј» Event Handlers

­ЪћД Event Handlers-рЃўрЃА рЃњрЃљрЃЏрЃЮрЃДрЃћрЃюрЃћрЃЉрЃљ

Inline Event Handler:

<button onclick=" alert('рЃдрЃўрЃџрЃљрЃЎрЃќрЃћ рЃЊрЃљрЃгрЃЎрЃљрЃърЃБрЃюрЃћрЃЉрЃљ!')">рЃЊрЃљрЃљрЃГрЃўрЃарЃћ</button>

JavaScript-рЃерЃў Event Handler:

const button = document.querySelector('#myButton');
button.onclick = function() {
    alert('рЃдрЃўрЃџрЃљрЃЎрЃќрЃћ рЃЊрЃљрЃгрЃЎрЃљрЃърЃБрЃюрЃћрЃЉрЃљ!');
};

Рџа№ИЈ Inline vs JavaScript Event Handlers

  • Inline: рЃЏрЃљрЃарЃбрЃўрЃЋрЃў, рЃЏрЃљрЃњрЃарЃљрЃЏ рЃљрЃа рЃљрЃарЃўрЃА рЃарЃћрЃЎрЃЮрЃЏрЃћрЃюрЃЊрЃћрЃЉрЃБрЃџрЃў рЃЊрЃўрЃЊрЃў рЃърЃарЃЮрЃћрЃЦрЃбрЃћрЃЉрЃўрЃАрЃЌрЃЋрЃўрЃА
  • JavaScript: рЃБрЃЎрЃћрЃЌрЃћрЃАрЃў рЃЎрЃЮрЃюрЃбрЃарЃЮрЃџрЃў рЃЊрЃљ рЃЏрЃарЃљрЃЋрЃљрЃџрЃ»рЃћрЃарЃљрЃЊрЃў Event Listener-рЃћрЃЉрЃўрЃА рЃЏрЃ«рЃљрЃарЃЊрЃљрЃГрЃћрЃарЃљ

рЃЏрЃарЃљрЃЋрЃљрЃџрЃ»рЃћрЃарЃљрЃЊрЃў Event Listener:

const button = document.querySelector('#myButton');
button.addEventListener('click', function() {
    console.log('рЃърЃўрЃарЃЋрЃћрЃџрЃў Listener');
});
button.addEventListener('click', function() {
    console.log('рЃЏрЃћрЃЮрЃарЃћ Listener');
});

­ЪџФ preventDefault

РЮЊ рЃарЃљ рЃљрЃарЃўрЃА preventDefault?

preventDefault() рЃЏрЃћрЃЌрЃЮрЃЊрЃў рЃАрЃљрЃерЃБрЃљрЃџрЃћрЃЉрЃљрЃА рЃўрЃФрЃџрЃћрЃЋрЃљ рЃЌрЃљрЃЋрЃўрЃЊрЃљрЃю рЃљрЃЋрЃўрЃфрЃўрЃџрЃЮрЃЌ рЃћрЃџрЃћрЃЏрЃћрЃюрЃбрЃўрЃА рЃюрЃљрЃњрЃБрЃџрЃўрЃАрЃ«рЃЏрЃћрЃЋрЃў рЃЦрЃЏрЃћрЃЊрЃћрЃЉрЃљ Event-рЃўрЃА рЃЊрЃарЃЮрЃА.

­ЪћД рЃњрЃљрЃЏрЃЮрЃДрЃћрЃюрЃћрЃЉрЃўрЃА рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃћрЃЉрЃў

рЃцрЃЮрЃарЃЏрЃўрЃА рЃњрЃљрЃњрЃќрЃљрЃЋрЃюрЃўрЃА рЃерЃћрЃЕрЃћрЃарЃћрЃЉрЃљ:

const form = document.querySelector('form');
form.addEventListener('submit', function(event) {
    event.preventDefault(); // рЃцрЃЮрЃарЃЏрЃўрЃА рЃюрЃљрЃњрЃБрЃџрЃўрЃАрЃ«рЃЏрЃћрЃЋрЃў рЃњрЃљрЃњрЃќрЃљрЃЋрЃюрЃўрЃА рЃерЃћрЃЕрЃћрЃарЃћрЃЉрЃљ
    console.log('рЃцрЃЮрЃарЃЏрЃљ рЃљрЃа рЃњрЃљрЃўрЃњрЃќрЃљрЃЋрЃюрЃљ!');
});

рЃџрЃўрЃюрЃЎрЃўрЃА рЃюрЃљрЃњрЃБрЃџрЃўрЃАрЃ«рЃЏрЃћрЃЋрЃў рЃЦрЃЏрЃћрЃЊрЃћрЃЉрЃўрЃА рЃерЃћрЃЕрЃћрЃарЃћрЃЉрЃљ:

const link = document.querySelector('a');
link.addEventListener('click', function(event) {
    event.preventDefault(); // рЃџрЃўрЃюрЃЎрЃўрЃА рЃюрЃљрЃњрЃБрЃџрЃўрЃАрЃ«рЃЏрЃћрЃЋрЃў рЃњрЃљрЃЊрЃљрЃЏрЃўрЃАрЃљрЃЏрЃљрЃарЃЌрЃћрЃЉрЃўрЃА рЃерЃћрЃЕрЃћрЃарЃћрЃЉрЃљ
    console.log('рЃџрЃўрЃюрЃЎрЃў рЃљрЃа рЃњрЃљрЃўрЃ«рЃАрЃюрЃљ!');
});

­Ъњ╗ рЃърЃарЃљрЃЦрЃбрЃўрЃЎрЃБрЃџрЃў рЃърЃарЃЮрЃћрЃЦрЃбрЃћрЃЉрЃў

­ЪЊЮ To-Do List рЃљрЃърЃџрЃўрЃЎрЃљрЃфрЃўрЃљ

рЃерЃћрЃЦрЃЏрЃћрЃюрЃўрЃЌ рЃЏрЃљрЃарЃбрЃўрЃЋрЃў To-Do List рЃљрЃърЃџрЃўрЃЎрЃљрЃфрЃўрЃљ, рЃАрЃљрЃЊрЃљрЃф рЃЏрЃЮрЃЏрЃ«рЃЏрЃљрЃарЃћрЃЉрЃћрЃџрЃў рЃерЃћрЃФрЃџрЃћрЃЉрЃА рЃЊрЃљрЃЋрЃљрЃџрЃћрЃЉрЃћрЃЉрЃўрЃА рЃЊрЃљрЃЏрЃљрЃбрЃћрЃЉрЃљрЃА, рЃгрЃљрЃерЃџрЃљрЃА рЃЊрЃљ рЃерЃћрЃАрЃарЃБрЃџрЃћрЃЉрЃўрЃА рЃЏрЃЮрЃюрЃўрЃерЃЋрЃюрЃљрЃА.

рЃўрЃюрЃАрЃбрЃарЃБрЃЦрЃфрЃўрЃћрЃЉрЃў:

  • HTML рЃцрЃЮрЃарЃЏрЃљ рЃљрЃ«рЃљрЃџрЃў рЃЊрЃљрЃЋрЃљрЃџрЃћрЃЉрЃўрЃА рЃерЃћрЃАрЃљрЃДрЃЋрЃљрЃюрЃљрЃЊ
  • рЃЊрЃљрЃЋрЃљрЃџрЃћрЃЉрЃћрЃЉрЃўрЃА рЃЊрЃўрЃюрЃљрЃЏрЃўрЃБрЃарЃў рЃЕрЃљрЃЏрЃЮрЃюрЃљрЃЌрЃЋрЃљрЃџрЃў
  • рЃЊрЃљрЃЋрЃљрЃџрЃћрЃЉрЃўрЃА рЃгрЃљрЃерЃџрЃўрЃА рЃЊрЃљ рЃерЃћрЃАрЃарЃБрЃџрЃћрЃЉрЃўрЃА рЃцрЃБрЃюрЃЦрЃфрЃўрЃЮрЃюрЃљрЃџрЃў
